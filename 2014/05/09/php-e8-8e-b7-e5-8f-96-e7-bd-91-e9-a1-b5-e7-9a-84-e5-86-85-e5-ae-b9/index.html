php获取网页的内容

By donahue

May 9 2014 更新日期:Mar 7 2015

文章目录

方法1：用file_get_contents，以get方式获取内容。

[php]
<?php
$url=’http://www.aituwo.com‘;
$html = file_get_contents($url);
echo $html;

[/php]

方法2：用fopen打开url，以get方式获取内容。

[php]
<?php
$url=’http://www.qiaonet.com;
$fp = fopen($url, ‘r’);
$header = stream_get_meta_data($fp);//获取报头信息
while(!feof($fp)) { $result .= fgets($fp, 1024); }
echo "url header: <br />;";
print_r($header);
echo " <br />;url body: $result";
fclose($fp);

[/php]

方法3：使用curl库，使用curl库之前，可能需要查看一下php.ini是否已经打开了curl扩展。

[php]
<?php
$ch = curl_init();
$timeout = 5;
curl_setopt ($ch, CURLOPT_URL,’http://www.nankezixunqq.com‘);
curl_setopt ($ch, CURLOPT_RETURNTRANSFER, 1);
curl_setopt ($ch, CURLOPT_CONNECTTIMEOUT, $timeout);
$file_contents = curl_exec($ch);
curl_close($ch);
echo $file_contents;

[/php]

方法4：用file_get_contents函数，以post方式获取url。

[php]
<?php
$url = ‘http://www.shouxufei.net‘;
$data = array (‘foo’ =>; ‘bar’);
$data = http_build_query($data);
$opts = array (
‘http’ =>; array (
‘method’ =>; ‘POST’,
‘header’=>; "Content-type: application/x-www-form-urlencoded\r\n"."Content-Length: " . strlen($data) . "\r\n",
‘content’ =>; $data
)
);
$ctx = stream_context_create($opts);
$html = @file_get_contents($url,’’,$ctx);
echo $html;

[/php]

$data = array (‘foo’ =>; ‘bar’, ‘bar’ =>; ‘baz’);

‘content’ =>; $data

方法6：用fsockopen函数打开url，获取完整的数据，包括header和body。

[php]
<?php
$url = ‘www.ttlxh.com’;
$fp = fsockopen($url,80,$errno,$errstr,30);
if (!$fp){
echo ‘error’;
} else {
$out = "GET / HTTP/1.1\r\n";
$out .= "Host: www.liulangtoo.com\r\n";
$out .= "Connection: Close\r\n\r\n";
fwrite($fp, $out);
while (!feof($fp)) {
echo fgets($fp, 128);
}
fclose($fp);
}

[/php]
